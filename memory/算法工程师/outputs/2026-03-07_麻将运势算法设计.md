```python
def get_time_ganzhi(hour):
    """根据小时获取时柱"""
    time_zodiac = ["子", "丑", "寅", "卯", "辰", "巳", "午", "未", "申", "酉", "戌", "亥"]
```

### 2.3 真太阳时校正公式
真太阳时 = 平太阳时 + 时差(Δ)

**时差计算公式（简化版）：**
```
Δ = 9.87 × sin(2B) - 7.53 × cos(B) - 1.5 × sin(B)
其中 B = (N - 81) × 360/365，N为年积日（1月1日为1）
```

**经度差修正：**
```
当地时间 = UTC时间 + (经度/15)小时
标准时区时间 = UTC时间 + 时区偏移
校正时间 = 当地时间 - 标准时区时间 + Δ
```

**Python实现：**
```python
import math
from datetime import datetime, timedelta

def true_solar_time_correction(date_obj, longitude, timezone_offset=8):
    """真太阳时校正
    
    Args:
        date_obj: datetime对象
        longitude: 经度（东经为正）
        timezone_offset: 时区偏移（小时）
    
    Returns:
        校正后的datetime对象
    """
    # 年积日计算
    year_start = datetime(date_obj.year, 1, 1)
    n = (date_obj - year_start).days + 1
    
    # 计算时差Δ
    B = math.radians((n - 81) * 360 / 365)
    delta_minutes = 9.87 * math.sin(2*B) - 7.53 * math.cos(B) - 1.5 * math.sin(B)
    
    # 经度差修正（东经120度为北京时间基准）
    longitude_correction = (longitude - 120) * 4  # 每度4分钟
    
    # 总校正时间（分钟）
    total_correction = delta_minutes + longitude_correction
    
    return date_obj + timedelta(minutes=total_correction)
```

```python
def calculate_wuxing_strength(bazi):
    """计算八字中五行强弱
    
    Args:
        bazi: 八字列表，如["甲子", "乙丑", "丙寅", "丁卯"]
    
    Returns:
        五行强度字典
    """
    wuxing_map = {
        "甲": "木", "乙": "木", "寅": "木", "卯": "木",
        "丙": "火", "丁": "火", "巳": "火", "午": "火",
        "戊": "土", "己": "土", "辰": "土", "戌": "土", "丑": "土", "未": "土",
        "庚": "金", "辛": "金", "申": "金", "酉": "金",
        "壬": "水", "癸": "水", "亥": "水", "子": "水"
    }
    
    strength = {"木": 0, "火": 0, "土": 0, "金": 0, "水": 0}
    
    for pillar in bazi:
        for char in pillar:
            if char in wuxing_map:
                element = wuxing_map[char]
                strength[element] += 1
    
    # 考虑地支藏干（简化：每个藏干加0.5分）
    hidden_map = {
        "子": ["癸"], "丑": ["己", "癸", "辛"], "寅": ["甲", "丙", "戊"],
        "卯": ["乙"], "辰": ["戊", "乙", "癸"], "巳": ["丙", "戊", "庚"],
        "午": ["丁", "己"], "未": ["己", "丁", "乙"], "申": ["庚", "壬", "戊"],
        "酉": ["辛"], "戌": ["戊", "辛", "丁"], "亥": ["壬", "甲"]
    }
    
    for pillar in bazi:
        for char in pillar:
            if char in hidden_map:
                for hidden in hidden_map[char]:
                    element = wuxing_map.get(hidden, "")
                    if element:
                        strength[element] += 0.5
    
    return strength
```

### 3.3 喜用神判断逻辑（简化版）
```python
def determine_xiyongshen(wuxing_strength):
    """判断喜用神（简化版）
    
    规则：
    1. 找出最弱的五行（需要补）
    2. 找出最强的五行（需要克或泄）
    3. 喜用神 = 生助最弱五行的五行
    
    Returns:
        喜用神列表
    """
    # 五行生克关系
    generate = {"木": "火", "火": "土", "土": "金", "金": "水", "水": "木"}
    overcome = {"木": "土", "火": "金", "土": "水", "金": "木", "水": "火"}
    
    # 找出最弱和最强的五行
    weakest = min(wuxing_strength, key=wuxing_strength.get)
    strongest = max(wuxing_strength, key=wuxing_strength.get)
    
    # 喜用神：生助最弱的五行，克制最强的五行
    likes = []
    
    # 生助最弱
    for element, target in generate.items():
        if target == weakest:
            likes.append(element)
    
    # 克制最强
    for element, target in overcome.items():
        if target == strongest:
            likes.append(element)
    
    return list(set(likes))  # 去重
```

## 4. 麻将运势评分算法

### 4.1 算法公式
```
运势评分 = 基础分(60) + 八字五行加成(-20~+20) + 当日干支加成(-15~+15) + 黄历宜忌加成(-10~+10)
```

### 4.2 八字五行加成计算
```python
def bazi_wuxing_bonus(bazi, xiyongshen):
    """八字五行加成（-20~+20）
    
    计算逻辑：
    1. 八字中喜用神出现次数越多，加成越高
    2. 忌神出现次数越多，扣分越多
    """
    # 假设已知忌神（与喜用神相克的五行）
    overcome = {"木": "金", "火": "水", "土": "木", "金": "火", "水": "土"}
    
    # 统计八字中各五行出现次数
    wuxing_count = calculate_wuxing_strength(bazi)
    
    # 计算加成
    bonus = 0
    for element in xiyongshen:
        count = wuxing_count.get(element, 0)
        bonus += count * 5  # 每个喜用神加5分
    
    # 忌神扣分
    for element in xiyongshen:
        ji_element = overcome.get(element)
        if ji_element:
            count = wuxing_count.get(ji_element, 0)
            bonus -= count * 3  # 每个忌神扣3分
    
    # 限制在-20~20之间
    return max(-20, min(20, bonus))
```

### 4.3 当日干支加成计算

```python
def daily_ganzhi_bonus(bazi, current_ganzhi):
```

### 4.4 黄历宜忌加成计算
```python
def huangli_bonus(huangli_yi, huangli_ji):
    """黄历宜忌加成（-10~+10）
    
    计算逻辑：
    1. 宜中有"祈福"、"求财"、"交易"等与财运相关：+5~+10
    2. 忌中有"赌博"、"交易"、"出行"等：-5~-10
    3. 其他：0
    """
    good_keywords = ["祈福", "求财", "交易", "开市", "纳财", "签约"]
    bad_keywords = ["赌博", "破财", "安葬", "出行", "动土"]
    
    bonus = 0
    
    # 检查宜的事项
    for yi in huangli_yi:
        for keyword in good_keywords:
            if keyword in yi:
                bonus += 2
    
    # 检查忌的事项
    for ji in huangli_ji:
        for keyword in bad_keywords:
            if keyword in ji:
                bonus -= 2
    
    # 限制在-10~10之间
    return max(-10, min(10, bonus))
```

## 5. 方位吉凶算法

### 5.1 方位↔五行对应

```python
        "虎": "猪", "猪": "虎",
```

### 7.1 System Prompt（心理咨询师人设）
```
你是一位专业的心理咨询师和传统文化爱好者，擅长用积极心理学结合传统智慧为用户提供生活建议。你的回答需要：
```

```python
#!/usr/bin/env python3
# -*- coding: utf-8 -*-
"""
麻将运势算法核心模块
"""

import math
import random
from datetime import datetime
from lunar import Lunar

class MahjongLuckCalculator:
    """麻将运势计算器"""
    
    def __init__(self):
        self.wuxing_map = {
            "甲": "木", "乙": "木", "寅": "木", "卯": "木",
            "丙": "火", "丁": "火", "巳": "火", "午": "火",
            "戊": "土", "己": "土", "辰": "土", "戌": "土", "丑": "土", "未": "土",
            "庚": "金", "辛": "金", "申": "金", "酉": "金",
            "壬": "水", "癸": "水", "亥": "水", "子": "水"
        }
        
        self.hidden_map = {
            "子": ["癸"], "丑": ["己", "癸", "辛"], "寅": ["甲", "丙", "戊"],
            "卯": ["乙"], "辰": ["戊", "乙", "癸"], "巳": ["丙", "戊", "庚"],
            "午": ["丁", "己"], "未": ["己", "丁", "乙"], "申": ["庚", "壬", "戊"],
            "酉": ["辛"], "戌": ["戊", "辛", "丁"], "亥": ["壬", "甲"]
        }
    
    def calculate_bazi(self, birth_date, birth_time):
        """计算八字"""
        lunar = Lunar.fromDate(birth_date)
        
        # 年柱、月柱、日柱
        year_pillar = lunar.getYearInGanZhi()
        month_pillar = lunar.getMonthInGanZhi()
        day_pillar = lunar.getDayInGanZhi()
        
        # 时柱
        hour = int(birth_time.split(":")[0])
        time_pillar = self._get_time_ganzhi(day_pillar, hour)
        
        return [year_pillar, month_pillar, day_pillar, time_pillar]
    
    def _get_time_ganzhi(self, day_pillar, hour):
        """计算时柱"""
        time_zodiac = ["子", "丑", "寅", "卯", "辰", "巳", "午", "未", "申", "酉", "戌", "亥"]
        
        if hour == 23 or hour == 0:
            idx = 0
        else:
            idx = (hour + 1) // 2
        
        # 日干推算时干（五鼠遁诀）
        day_gan = day_pillar[0]
        gan_list = ["甲", "乙", "丙", "丁", "戊", "己", "庚", "辛", "壬", "癸"]
        
        start_gan_idx = gan_list.index(day_gan)
        time_gan_idx = (start_gan_idx * 2 + idx) % 10
        
        return f"{gan_list[time_gan_idx]}{time_zodiac[idx]}"
    
    def calculate_wuxing_strength(self, bazi):
        """计算五行强度"""
        strength = {"木": 0, "火": 0, "土": 0, "金": 0, "水": 0}
        
        # 统计天干地支
        for pillar in bazi:
            for char in pillar:
                if char in self.wuxing_map:
                    element = self.wuxing_map[char]
                    strength[element] += 1
        
        # 统计藏干
        for pillar in bazi:
            for char in pillar:
                if char in self.hidden_map:
                    for hidden in self.hidden_map[char]:
                        element = self.wuxing_map.get(hidden, "")
                        if element:
                            strength[element] += 0.3
        
        return strength
    
    def determine_xiyongshen(self, wuxing_strength):
        """判断喜用神"""
        generate = {"木": "火", "火": "土", "土": "金", "金": "水", "水": "木"}
        overcome = {"木": "土", "火": "金", "土": "水", "金": "木", "水": "火"}
        
        weakest = min(wuxing_strength, key=wuxing_strength.get)
        strongest = max(wuxing_strength, key=wuxing_strength.get)
        
        likes = []
        
        # 生助最弱
        for element, target in generate.items():
            if target == weakest:
                likes.append(element)
        
        # 克制最强
        for element, target in overcome.items():
            if target == strongest:
                likes.append(element)
        
        return list(set(likes))
    
    def calculate_luck_score(self, bazi, current_date):
        """计算运势评分"""
        base_score = 60
        
        # 八字五行加成
        wuxing_strength = self.calculate_wuxing_strength(bazi)
        xiyongshen = self.determine_xiyongshen(wuxing_strength)
        bazi_bonus = self._bazi_bonus(bazi, xiyongshen)
        
        # 当日干支加成
        current_lunar = Lunar.fromDate(current_date)
        current_ganzhi = current_lunar.getDayInGanZhi()
        daily_bonus = self._daily_bonus(bazi[2], current_ganzhi)
        
        # 黄历宜忌加成（模拟数据）
        huangli_bonus_val = random.randint(-5, 5)
        
        total = base_score + bazi_bonus + daily_bonus + huangli_bonus_val
        return max(0, min(100, int(total)))
    
    def _bazi_bonus(self, bazi, xiyongshen):
        """八字五行加成"""
        overcome = {"木": "金", "火": "水", "土": "木", "金": "火", "水": "土"}
        
        wuxing_count = self.calculate_wuxing_strength(bazi)
        
        bonus = 0
        for element in xiyongshen:
            count = wuxing_count.get(element, 0)
            bonus += count * 4
        
        for element in xiyongshen:
            ji_element = overcome.get(element)
            if ji_element:
                count = wuxing_count.get(ji_element, 0)
                bonus -= count * 2
        
        return max(-20, min(20, bonus))
    
    def _daily_bonus(self, day_pillar, current_ganzhi):
        """当日干支加成"""
        liuhe = {"子": "丑", "丑": "子", "寅": "亥", "亥": "寅",
                "卯": "戌", "戌": "卯", "辰": "酉", "酉": "辰",
                "巳": "申", "申": "巳", "午": "未", "未": "午"}
        
        liuchong = {"子": "午", "午": "子", "丑": "未", "未": "丑",
                   "寅": "申", "申": "寅", "卯": "酉", "酉": "卯",
                   "辰": "戌", "戌": "辰", "巳": "亥", "亥": "巳"}
        
        if current_ganzhi == day_pillar:
            return 15
        elif liuhe.get(day_pillar[1]) == current_ganzhi[1]:
            return 10
        elif liuchong.get(day_pillar[1]) == current_ganzhi[1]:
            return -10
        else:
            return random.randint(-5, 5)
    
    def get_lucky_colors(self, xiyongshen):
        """获取幸运颜色"""
        element_to_colors = {
            "木": ["青色", "绿色"],
            "火": ["红色", "紫色"],
            "土": ["黄色", "棕色"],
            "金": ["白色", "金色"],
            "水": ["黑色", "蓝色"]
        }
        
        colors = []
        for element in xiyongshen[:2]:  # 取前两个喜用神
            colors.extend(element_to_colors.get(element, []))
        
        return list(set(colors))[:4]  # 去重，最多4种
    
    def get_lucky_numbers(self, xiyongshen):
        """获取幸运数字"""
        element_to_numbers = {
            "木": [3, 8, 13, 18],
            "火": [2, 7, 12, 17],
            "土": [5, 0, 10, 15],
            "金": [4, 9, 14, 19],
            "水": [1, 6, 11, 16]
        }
        
        numbers = []
        for element in xiyongshen[:2]:
            numbers.extend(element_to_numbers.get(element, []))
        
        return list(set(numbers))[:4]


def main():
    """示例使用"""
    calculator = MahjongLuckCalculator()
    
    # 用户信息

if __name__ == "__main__":
    main()
```
