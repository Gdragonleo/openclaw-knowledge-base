## 1. 需求概述

根据用户生辰八字，分析用户今日/本周适合打哪种花色（万、筒、条），哪种牌型。

## 2. 五行→花色映射

| 五行 | 花色 | 代表色 | 备注 |
|------|------|--------|------|
| 木 | 条 | 绿色、青色 | 竹牌 |
| 火 | 筒 | 红色、紫色 | 索子 |
| 土 | 万 | 黄色、棕色 | 萬子 |
| 金 | 筒/条 | 白色、金色 | 特殊调整 |
| 水 | 条 | 黑色、蓝色 | 特殊调整 |

## 3. 算法逻辑

```python
def calculate_suit_luck(bazi, xiyongshen, current_ganzhi):
    """计算花色运势
    
    Args:
        bazi: 八字列表
        xiyongshen: 喜用神列表
        current_ganzhi: 当日干支
    
    Returns:
        各花色运势评分
    """
    # 五行→花色映射
    wuxing_to_suit = {
        "木": {"suit": "条", "score_bonus": 15},
        "火": {"suit": "筒", "score_bonus": 15},
        "土": {"suit": "万", "score_bonus": 15},
        "金": {"suit": "筒/条", "score_bonus": 10},  # 金生水，条也可
        "水": {"suit": "条", "score_bonus": 12}
    }
    
    # 基础分
    base_scores = {"万": 60, "筒": 60, "条": 60}
    
    # 喜用神加成
    for element in xiyongshen:
        if element in wuxing_to_suit:
            suit = wuxing_to_suit[element]["suit"]
            bonus = wuxing_to_suit[element]["score_bonus"]
            base_scores[suit] += bonus
    
    # 忌神扣分
    overcome = {"木": "金", "火": "水", "土": "木", "金": "火", "水": "土"}
    for element in xiyongshen:
        ji_element = overcome.get(element, "")
        if ji_element in wuxing_to_suit:
            suit = wuxing_to_suit[ji_element]["suit"]
            base_scores[suit] -= 8
    
    # 当日干支微调
    ganzhi_wuxing = get_ganzhi_wuxing(current_ganzhi)
    if ganzhi_wuxing in wuxing_to_suit:
        suit = wuxing_to_suit[ganzhi_wuxing]["suit"]
        base_scores[suit] += 5
    
    # 归一化到0-100
    for suit in base_scores:
        base_scores[suit] = max(0, min(100, base_scores[suit]))
    
    # 排序返回
    sorted_suits = sorted(base_scores.items(), key=lambda x: x[1], reverse=True)
    
    return {
        "rankings": sorted_suits,
        "best_suit": sorted_suits[0][0],
        "best_score": sorted_suits[0][1],
        "scores": base_scores
    }
```

## 4. 完整代码示例

```python
#!/usr/bin/env python3
# 麻将花色运势计算器

class MahjongSuitCalculator:
    def __init__(self):
        self.wuxing_map = {
            "甲": "木", "乙": "木", "寅": "木", "卯": "木",
            "丙": "火", "丁": "火", "巳": "火", "午": "火",
            "戊": "土", "己": "土", "辰": "土", "戌": "土", "丑": "土", "未": "土",
            "庚": "金", "辛": "金", "申": "金", "酉": "金",
            "壬": "水", "癸": "水", "亥": "水", "子": "水"
        }
        
        self.wuxing_suit = {
            "木": {"suit": "条", "bonus": 15},
            "火": {"suit": "筒", "bonus": 15},
            "土": {"suit": "万", "bonus": 15},
            "金": {"suit": "筒", "bonus": 10},
            "水": {"suit": "条", "bonus": 12}
        }
    
    def get_suit_luck(self, bazi, xiyongshen, current_ganzhi):
        """计算花色运势"""
        scores = {"万": 60, "筒": 60, "条": 60}
        
        # 喜用神加成
        for element in xiyongshen:
            if element in self.wuxing_suit:
                suit = self.wuxing_suit[element]["suit"]
                scores[suit] += self.wuxing_suit[element]["bonus"]
        
        # 忌神扣分
        overcome = {"木": "金", "火": "水", "土": "木", "金": "火", "水": "土"}
        for element in xiyongshen:
            ji = overcome.get(element, "")
            if ji in self.wuxing_suit:
                suit = self.wuxing_suit[ji]["suit"]
                scores[suit] -= 8
        
        # 当日干支微调
        if current_ganzhi:
            ganzhi_wuxing = self.wuxing_map.get(current_ganzhi[0], "")
            if ganzhi_wuxing in self.wuxing_suit:
                suit = self.wuxing_suit[ganzhi_wuxing]["suit"]
                scores[suit] += 5
        
        # 归一化
        for suit in scores:
            scores[suit] = max(0, min(100, scores[suit]))
        
        # 排序
        rankings = sorted(scores.items(), key=lambda x: x[1], reverse=True)
        
        return {
            "rankings": rankings,
            "best_suit": rankings[0][0],
            "best_score": rankings[0][1],
            "all_scores": scores
        }

# 使用示例
calculator = MahjongSuitCalculator()
bazi = ["庚午", "戊子", "丙寅", "甲午"]
xiyongshen = ["木", "火"]
current_ganzhi = "壬申"

result = calculator.get_suit_luck(bazi, xiyongshen, current_ganzhi)
print(f"最佳花色: {result['best_suit']} ({result['best_score']}分)")
print(f"运势排名: {result['rankings']}")
```
