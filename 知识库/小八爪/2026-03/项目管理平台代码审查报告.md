```java
package com.cqiic.orange.project.service;

import org.springframework.stereotype.Service;

/**
 * 数据迁移和兼容服务
 */
@Service
public class DataMigrationService {
    
    /**
     * 兼容历史数据
     */
    public ProjectVO adaptProjectData(Object rawData) {
        ProjectVO vo = new ProjectVO();
        
        // 判断数据来源
        if (isOldData(rawData)) {
            // 旧数据结构适配
            adaptOldData(rawData, vo);
        } else {
            // 新数据结构
            adaptNewData(rawData, vo);
        }
        
        return vo;
    }
    
    private boolean isOldData(Object data) {
        // 判断逻辑：根据字段名或数据结构
        return data instanceof Map && ((Map) data).containsKey("project_id");
    }
    
    private void adaptOldData(Object data, ProjectVO vo) {
        Map<String, Object> oldData = (Map<String, Object>) data;
        vo.setId((String) oldData.get("project_id"));
        vo.setName((String) oldData.get("project_name"));
        // ... 其他字段映射
    }
    
    private void adaptNewData(Object data, ProjectVO vo) {
        BeanUtils.copyProperties(data, vo);
    }
}
```

### 修改4: 前端数据监听和防覆盖

#### 创建 `useProjectData.ts`
```typescript
import { ref, watch, computed } from 'vue'
import { projectApi } from '@/api/project'

export function useProjectData(projectId: string) {
  const project = ref<Project>({} as Project)
  const loading = ref(false)
  const error = ref<Error | null>(null)
  
  // 加载项目数据
  const loadProject = async () => {
    loading.value = true
    error.value = null
    
    try {
      const data = await projectApi.detail(projectId)
      // 深拷贝防止引用污染
      project.value = JSON.parse(JSON.stringify(data))
    } catch (e) {
      error.value = e as Error
    } finally {
      loading.value = false
    }
  }
  
  // 监听数据变化
  watch(project, (newVal, oldVal) => {
    // 检测是否被意外覆盖
    if (oldVal && Object.keys(oldVal).length > 0 && Object.keys(newVal).length === 0) {
      console.warn('项目数据可能被意外清空', { old: oldVal, new: newVal })
    }
  }, { deep: true })
  
  // 更新数据（防止覆盖关键字段）
  const updateProject = async (updates: Partial<Project>) => {
    const originalData = { ...project.value }
    
    try {
      // 合并更新而非替换
      const mergedData = { ...project.value, ...updates }
      await projectApi.update(projectId, mergedData)
      project.value = mergedData
    } catch (e) {
      // 回滚
      project.value = originalData
      throw e
    }
```
