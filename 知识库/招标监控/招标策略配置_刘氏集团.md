```python
    """
```

```python
def filter_projects(项目):
    """
    一级筛选：必须满足
    """
    # 1. 项目类型
    if 项目.类型 not in ['勘察', '设计', '监测', '测绘']:
        return False
    
    # 2. 资质匹配
    if 项目.资质要求 > 我们的资质:
        return False
    
    # 3. 地域限制
    if 项目.地区 not in ['重庆', '西南地区']:
        return False
    
    return True
```

```python
def score_projects(项目):

        分数 += 15

    if 项目.甲方 in ['政府', '国企']:

        分数 += 10
    
    # 6. 本地化优势（+10分）
```
